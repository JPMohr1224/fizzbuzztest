 public static string fizzbuzz(int number)
    {

        if (number % 5 == 0 && number % 3 == 0)
        {
            string x = "fizzbuzz";
            return x;
        }
        else if (number % 5 == 0)
        {
            string x = "buzz";
            return x;
        }
        else number % 3 == 0
        {
            string x = "fizz";
            return x;
        }
